#include<iostream>
#include<random>

int main() {
	int* arr = new int[23];

	std::mt19937 gen(12313421);
	std::uniform_int_distribution<int> dist(-100, 100);
	for (int i = 0; i < 22; i++) {
		arr[i] = dist(gen);
	}
	arr[22] = 90;
	for (int i = 0; i < 23; i++) {
		std::cout << arr[i] << " \n";
	};
	__asm {
        mov esi, arr
        mov ecx, 23
        dec ecx
        outerLoop :
            mov ebx, 0
            mov edi, 0
            innerloop :
            mov eax, [esi + edi * 4]
            mov edx, [esi + edi * 4 + 4]

   
        cmp eax, edx
            jle noSwap

            mov eax, [esi + edi * 4]
            mov edx, [esi + edi * 4 + 4]
            mov[esi + edi * 4], edx
            mov[esi + edi * 4 + 4], eax

            mov ebx, 1

            noSwap:
        inc edi
            cmp edi, 23
            jl innerLoop

            cmp ebx, 0
            je outerLoopEnd
            loop outerLoop
            outerLoopEnd :
	}
	std::cout << "\n";
	for (int i = 0; i < 23; i++) {
		std::cout << arr[i] << " \n";
	};
	system("pause");
}
